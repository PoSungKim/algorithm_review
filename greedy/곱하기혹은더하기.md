### 문제: 곱하기혹은더하기

```C++
#include <algorithm>
#include <iostream>
#include <string>
#include <vector>
using namespace std;

string input;
int main(void)
{
    cin >> input;
    int curNum = input[0] - '0';
    for (int i = 1; i < input.length(); i++)
    {
        int nextNum = input[i] - '0';
        (curNum <= 1 || nextNum <= 1) 
            ? curNum += nextNum  // 현재 숫자 혹은 다음 숫자가 0 또는 1이라면 더하기
            : curNum *= nextNum; // 현재 숫자 혹은 다음 숫자가 2 이상이면 더하기
    }
    cout << curNum << endl;
    return 0;
}
```

* 0을 곱하면 0
* 1을 곱하면 자기 자신
