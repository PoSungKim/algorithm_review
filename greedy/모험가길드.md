```C++
#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;

int list[100001], N, cnt = 0;
int main(void)
{
    cin >> N;
    for (int i = 0; i < N; i++)
        cin >> list[N];
        
    sort(list, list + N);
    
    for (int i = 1;; i++)
    {
        N -= i;
        if (N < 0)
            break;
        cnt++;
    }
    
    cout << cnt << endl;
    return 0;
}
```
